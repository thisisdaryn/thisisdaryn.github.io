Sample R Map

library(sf)
library(tidyverse)
library(rvest)
library(readxl)

us_map <- read_sf("gz_2010_us_040_00_5m/gz_2010_us_040_00_5m.shp") 

lower_48 <- us_map %>% filter(!(NAME %in% c("Alaska", "Hawaii", "Puerto Rico")))

pop_table <- read_xlsx("nst-est2019-01.xlsx", skip = 3,
                       sheet = "NST01")
names(pop_table)[1] <- "Geo_Area"

pop_table <- pop_table %>% mutate(Geo_Area = str_remove(Geo_Area, "\\."))

pop_map <- left_join(lower_48, pop_table, by = c("NAME" = "Geo_Area"))

centroids <- pop_map %>% st_centroid() 

ggplot() + 
  geom_sf(data = pop_map) + 
  geom_sf(data = centroids, aes(size = Census)) + 
  theme(legend.position = "none")
