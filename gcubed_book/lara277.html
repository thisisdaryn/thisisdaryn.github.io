Data Manipulation

  • Separate test matches from ODIs (One-day Internationals)
  • Rank the Opponents by the total number of runs scored against each
library(dplyr) # to use filter, group_by, summarise and arrange
library(tidyr) # to use the separate function

lara_tests <- lara %>% separate(Match, into = c("Format", "MatchNum")) %>% 
  filter(Format == "Test")

#Order opponents by total runs scored against: 
lara_test_summary <- group_by(lara_tests, Opp) %>% 
  summarise(Runs = sum(Runs, na.rm = TRUE)) %>% 
  arrange(desc(Runs))

lara_test_summary$Opp <- factor(lara_test_summary$Opp, levels = lara_test_summary$Opp)

library(ggplot2)
library(scales) #to get commas in formatting numerical values on the y-axis

bcl_runs_plt <- ggplot(lara_test_summary, aes(x = Opp, y = Runs)) + 
  geom_bar(stat = "identity", fill = "mediumpurple", colour = "black") +
  xlab("Opponent") + 
  ggtitle("Brian Lara Test Career (1990-2006)")+
  scale_y_continuous(label = comma)+ 
  theme( plot.title = element_text(size = 16, face = "bold", hjust = 0.5))
