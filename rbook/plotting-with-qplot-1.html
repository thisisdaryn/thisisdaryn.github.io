8.2.1 Histograms

qplot(x = AnnualSalary, data = large_depts, geom = "histogram") 

qplot(x = AnnualSalary, data = large_depts, geom = "histogram", fill = I("orange"), colour = I("black")) 

library(ggplot2)
qplot(x = AnnualSalary, data = large_depts, geom = "histogram", facets = Department~.)

qplot(x = AnnualSalary, data = large_depts, geom = "histogram", facets = Department~., scale = "free_y") 

qplot(x = AnnualSalary, data = large_depts, geom = "histogram", fill = Department) + 
  facet_wrap(Department~., scales = "free_y", ncol = 1)  

qplot(x = AnnualSalary, data = large_depts, geom = "histogram", fill = Department, colour = I("black")) + 
  facet_wrap(Department~., scales = "free_y", ncol = 1)  

qplot(x = AnnualSalary, data = large_depts, geom = "density", fill = Department, alpha = I(0.3))

8.2.2 Density plots

qplot(x = AnnualSalary, data = large_depts, geom = "density", fill = Department, alpha = I(0.3)) + 
  facet_wrap(Department~., ncol = 1)

8.2.3 Box plots

qplot(x = Department, y = AnnualSalary, data = large_depts, geom = "boxplot")

8.3.1 Time Series

apple_prod$X = paste(apple_prod$Year, apple_prod$Quarter)
qplot(x = X, y = Revenue, data = apple_prod, group = Product, geom = c("point", "line"))

apple_prod$X = paste(apple_prod$Year, apple_prod$Quarter, sep = " Q")
qplot(x = X, y = Revenue, data = apple_prod, group = Product, colour = Product, geom = c("point", "line")) 

df <- apple_prod
df$Quarter = paste(apple_prod$Year, apple_prod$Quarter, sep = " Q")
